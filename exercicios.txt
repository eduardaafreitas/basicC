        if(v[i] > maior)

    }
    maior++;
    int aux[maior];
    for(i = 0; i < maior; i++){
        aux[i] = 0;
    }
    for(i = 0; i < n; i++){
        aux[v[i]]++;
    }
    printf("vetor original: \n");

    }
    printf("\nrepetições: \n");
    for(i = 0; i < maior; i++){
        if (aux[i] != 0)
            printf("elemento %d apareceu %d vezes \n", i, aux[i]);

#include <stdio.h> 

int main (){
    int n;
    printf("Entre com o valor de N: ");
    scanf("%d", &n);
    if(n > 1024) {
        printf("numero invalido \n");
        return 0;
    }

    int v[n];
    int v2[n];
    int conc[n * 2];
    int i;


    for(i = 0; i < n; i++){
        scanf("%d", &v[i]);
        v2[i] = v[i];
    }

    for(i = 0; i < n; i++){
        conc[i] = v[i];
    }

    for(int i = 0;i < (2*n);i++){
        if(i < n)
            conc[i] = v[i];
        else
            conc[i] = v2[i - n];
    }

    printf("\n vetor original: \n");
    for(i = 0; i < n; i++){
        printf("%d ", v[i]);
    }

    printf("\n vetor 2: \n");
    for(i = 0; i < n; i++){
        printf("%d ", v2[i]);
    }

    printf("\n vetores concatenados: \n");
    for(i = 0; i < n * 2; i++){
        printf("%d ", conc[i]);
    }
    printf("\n");
    return 0;
}

    int v[n];

    int par = 0;
    int impar = 0;
    
    for(i = 0; i < n; i++){

        else
